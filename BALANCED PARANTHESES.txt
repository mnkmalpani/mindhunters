def isBalanaced(s):
stack = [ ]
brackets = {‘(‘ : ’)’,’{‘ : ‘}’, ‘[‘ : ‘]’ }
for char in s:
if char in [‘(‘ , ‘{‘ , ‘[‘ ]:
stack.append(char)
else:
if stack:
top = stack.pop()
if brackets[top] != char:
return “NO”
else:
return “NO”

if stack:
return “NO”
else:
return “YES”
